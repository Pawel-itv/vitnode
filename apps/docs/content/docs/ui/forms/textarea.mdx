## Usage

import { Tab, Tabs } from 'fumadocs-ui/components/tabs';

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">
```ts
import * as z from 'zod';
import { AutoForm } from 'vitnode-frontend/form/auto-form';
```

```ts
const formSchema = z.object({
  post: z.string(),
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fieldConfig={{
    post: {
      label: 'Post',
      fieldType: AutoFormTextArea,
    },
  }}
/>
```

  </Tab>

### With placeholder

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">
```ts
import * as z from 'zod';
import { AutoForm } from 'vitnode-frontend/form/auto-form';
```

```ts
const formSchema = z.object({
  post: z.string(),
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fieldConfig={{
    post: {
      label: 'Post',
      fieldType: props => (
        <AutoFormTextArea {...props} placeholder="Type your thoughts here..." /> // [!code highlight]
      ),
    },
  }}
/>
```

  </Tab>

  <Tab value="Manual">
```ts
import { Textarea } from 'vitnode-frontend/components/ui/textarea';
```

```tsx
<Textarea placeholder="Type your thoughts here..." />
```

  </Tab>
</Tabs>
