## Usage

import { Tab, Tabs } from 'fumadocs-ui/components/tabs';

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">
```ts
import * as z from 'zod';
import { AutoForm } from 'vitnode-frontend/components/ui/auto-form';
```

```ts
const formSchema = z.object({
  username: z.string(),
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fieldConfig={{
    username: {
      label: 'Username',
    },
  }}
/>
```

  </Tab>

  <Tab value="Manual">
```ts
import { Input } from 'vitnode-frontend/components/ui/input';
```

```tsx
<Input />
```

  </Tab>
</Tabs>

### With placeholder

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">

```tsx
<AutoForm
  formSchema={formSchema}
  fieldConfig={{
    username: {
      label: 'Username',
      inputProps: {
        placeholder: 'Username Placeholder', // [!code highlight]
      },
    },
  }}
/>
```

  </Tab>

  <Tab value="Manual">

```tsx
<Input placeholder="Email" />
```

  </Tab>
</Tabs>

### Number Input

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">

```ts

  count: z.coerce.number(), // [!code highlight]
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fieldConfig={{
    count: {
      label: 'Count',
      inputProps: {
        type: 'number', // [!code highlight]
      },
    },
  }}
/>
```

  </Tab>

  <Tab value="Manual">
