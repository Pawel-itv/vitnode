## Usage

import { Tab, Tabs } from 'fumadocs-ui/components/tabs';

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">
```ts
import * as z from 'zod';
import { AutoForm } from 'vitnode-frontend/form/auto-form';
import { AutoFormInput } from 'vitnode-frontend/form/fields/input';
```

```ts
const formSchema = z.object({
  username: z.string(),
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fieldConfig={{
    username: {
      label: 'Username',
      fieldType: AutoFormInput, // [!code highlight]
    },
  }}
/>
```

  </Tab>

  <Tab value="Manual">
```ts
import { Input } from 'vitnode-frontend/components/ui/input';
```

```tsx
<Input />
```

  </Tab>
</Tabs>

### With placeholder

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">

```tsx
<AutoForm
  formSchema={formSchema}
  fieldConfig={{
    username: {
      label: 'Username',
      fieldType: props => (
        <AutoFormInput placeholder="Type username" {...props} /> // [!code highlight]
      ),
    },
  }}
/>
```

  </Tab>

  <Tab value="Manual">

```tsx
<Input placeholder="Email" />
```

  </Tab>
</Tabs>
