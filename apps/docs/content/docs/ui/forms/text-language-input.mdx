## Usage

import { Tab, Tabs } from 'fumadocs-ui/components/tabs';

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">
```ts
import * as z from 'zod';
import { zodLanguageInput } from 'vitnode-frontend/helpers/zod';
import { AutoForm } from 'vitnode-frontend/components/ui/auto-form';
import { AutoFormTextLanguageInput } from 'vitnode-frontend/components/ui/auto-form/fields/text-language-input';
```

```ts
const formSchema = z.object({
  description: z.array(zodLanguageInput),
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fieldConfig={{
    description: {
      label: 'Description',
      fieldType: AutoFormTextLanguageInput,
    },
  }}
/>
```

  </Tab>

  <Tab value="Manual">
```ts
import { TextLanguageInput } from 'vitnode-frontend/components/ui/text-language-input'; 
```

```tsx
<TextLanguageInput onChange={onChange} value={value} />
```

  </Tab>
</Tabs>
