## Usage

import { Tab, Tabs } from 'fumadocs-ui/components/tabs';

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">
```ts
import * as z from 'zod';
import { AutoForm } from 'vitnode-frontend/components/ui/auto-form';
```

```ts
const formSchema = z.object({
  color_primary: z.string(),
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fieldConfig={{
    color_primary: {
      label: 'Color picker',
      fieldType: 'color',
      inputProps: {
        disableRemoveColor: true,
      },
    },
  }}
/>
```

  </Tab>

  <Tab value="Manual">
```tsx
import { ColorPicker } from 'vitnode-frontend/components/ui/color-picker';
```

```tsx
<ColorPicker onChange={onChange} value={value} />
```

  </Tab>
</Tabs>

  the `fieldConfig` object.
