## Usage

import { Tab, Tabs } from 'fumadocs-ui/components/tabs';

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">
```ts
import * as z from 'zod';
import { AutoForm } from 'vitnode-frontend/components/ui/auto-form';
```

```ts
const formSchema = z.object({
  allow_files: z.boolean(),
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fieldConfig={{
    allow_files: {
      label: 'Allow Files',
      inputProps: {},
    },
  }}
/>
```

  </Tab>

  <Tab value="Manual">
```ts
import { Switch } from 'vitnode-frontend/components/ui/switch';
```

```tsx
<Switch checked={value} onCheckedChange={onChange} />
```

  </Tab>
</Tabs>
