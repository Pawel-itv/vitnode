title: Tag Input
description: Tag Input is a component that allows users to input tags.
---

## Usage

import { Tab, Tabs } from 'fumadocs-ui/components/tabs';

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">
```ts
import * as z from 'zod';
import { AutoForm } from 'vitnode-frontend/components/ui/auto-form';
import { AutoFormTags } from 'vitnode-frontend/components/ui/auto-form/fields/tags';
import { zodTag } from 'vitnode-frontend/helpers/zod';
```

```ts
const formSchema = z.object({
  keyworlds: zodTag,
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fieldConfig={{
    keyworlds: {
      label: 'Keyworlds',
      fieldType: AutoFormTags, // [!code highlight]
    },
  }}
/>
```

  </Tab>

  <Tab value="Manual">
```ts
import { TagsInput } from 'vitnode-frontend/components/ui/tags-input';
```

```tsx
<TagsInput onChange={onChange} value={value} />
```

  </Tab>
</Tabs>

### Multiple

```tsx
<AutoForm
  formSchema={formSchema}
  fieldConfig={{
    keyworlds: {
      label: 'Keyworlds',
      fieldType: props => <AutoFormTags {...props} multiple />, // [!code highlight]
    },
  }}
/>
```
