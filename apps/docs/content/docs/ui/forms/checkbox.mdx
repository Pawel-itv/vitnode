## Usage

import { Tab, Tabs } from 'fumadocs-ui/components/tabs';

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">
```ts
import * as z from 'zod';
import { AutoForm } from 'vitnode-frontend/components/ui/auto-form';
import { AutoFormCheckbox } from 'vitnode-frontend/components/ui/auto-form/fields/checkbox';
```

```ts
const formSchema = z.object({
  allow_files: z.boolean(),
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fieldConfig={{
    allow_files: {
      label: 'Allow Files',
      fieldType: AutoFormCheckbox, // [!code highlight]
    },
  }}
/>
```

  </Tab>

  <Tab value="Manual">
```ts
import { Checkbox } from 'vitnode-frontend/components/ui/checkbox';
```

```tsx
<Checkbox onChange={onChange} checked={checked} />
```

  </Tab>
</Tabs>
