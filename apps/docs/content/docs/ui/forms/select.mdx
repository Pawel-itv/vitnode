## Usage

import { Tab, Tabs } from 'fumadocs-ui/components/tabs';

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">
```ts
import * as z from 'zod';
import { AutoForm } from 'vitnode-frontend/components/ui/auto-form';
```

```ts
const formSchema = z.object({
  type: z.enum(['smtp', 'resend', 'none']),
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fieldConfig={{
    type: {
      label: 'Type of Email',
      inputProps: {
        placeholder: 'Select type of email',
      },
    },
  }}
/>
```

### With Native Enum

```ts
import * as z from 'zod';
import { AutoForm } from 'vitnode-frontend/components/ui/auto-form';
```

```ts
const EmailProvider = {
  none: 'none',
  resend: 'resend',
  smtp: 'smtp',
} as const;

// or

enum EmailProvider {
  none = 'none',
  resend = 'resend',
  smtp = 'smtp',
}
```

```ts
const formSchema = z.object({
  type: z.nativeEnum(EmailProvider), // [!code highlight]
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fieldConfig={{
    type: {
      label: 'Type of Email',
      inputProps: {
        placeholder: 'Select type of email',
      },
    },
  }}
/>
```

### With Translations

```ts
import * as z from 'zod';
import { useTranslations } from 'next-intl'; // [!code highlight]
import { AutoForm } from 'vitnode-frontend/components/ui/auto-form';
```

```ts
const t = useTranslations('admin.core.settings.email'); // [!code highlight]

const EmailProvider = {
  none: t('provider.none'),
  resend: t('provider.resend'),
  smtp: t('provider.smtp'),
} as const;
```

```ts
const formSchema = z.object({
  type: z.nativeEnum(EmailProvider), // [!code highlight]
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fieldConfig={{
    type: {
      label: 'Type of Email',
      inputProps: {
        placeholder: 'Select type of email',
      },
    },
  }}
/>
```
