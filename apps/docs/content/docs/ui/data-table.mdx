## Usage

```tsx
import { DataTable } from 'vitnode-frontend/components/ui/data-table';
```

```tsx
<DataTable
  data={[
    { id: 1, name: 'John Doe' },
    { id: 2, name: 'Jane Doe 2' },
  ]}
  columns={[
    { title: 'ID', id: 'id' },
    { title: 'Name', id: 'name' },
  ]}
  defaultSorting={{
    sortBy: 'id',
    sortDirection: 'desc',
  }}
/>
```

## Examples

### With Pagination

```tsx
<DataTable
  data={[
    { id: 1, name: 'John Doe' },
    { id: 2, name: 'Jane Doe 2' },
  ]}
  columns={[
    { title: 'ID', id: 'id' },
    { title: 'Name', id: 'name' },
  ]}
  defaultSorting={{
    sortBy: 'id',
    sortDirection: 'desc',
  }}
  // [!code word:pageInfo]
  pageInfo={{
    count: 2,
    endCursor: 2,
    hasNextPage: false,
    hasPreviousPage: false,
    startCursor: 1,
    totalCount: 2,
  }}
/>
```

### With Search

```tsx
<DataTable
  data={[
    { id: 1, name: 'John Doe' },
    { id: 2, name: 'Jane Doe 2' },
  ]}
  columns={[
    { title: 'ID', id: 'id' },
    { title: 'Name', id: 'name' },
  ]}
  defaultSorting={{
    sortBy: 'id',
    sortDirection: 'desc',
  }}
  searchPlaceholder="Search users..." // [!code highlight]
/>
```

### With Custom Cell

```tsx
<DataTable
  data={[
    { id: 1, name: 'John Doe' },
    { id: 2, name: 'Jane Doe 2' },
  ]}
  columns={[
    { title: 'ID', id: 'id' },
    {
      title: 'Name',
      id: 'name',
      // [!code word:cell]
      cell: ({ row }) => {
        return <span className="font-bold">{row.name}</span>;
      },
    },
  ]}
  defaultSorting={{
    sortBy: 'id',
    sortDirection: 'desc',
  }}
/>
```

#### Actions

We created special column for actions. You can add buttons, links or any other components. Flex is used for the layout with gap.

```tsx
<DataTable
  data={[
    { id: 1, name: 'John Doe' },
    { id: 2, name: 'Jane Doe 2' },
  ]}
  columns={[
    { title: 'ID', id: 'id' },
    { title: 'Name', id: 'name' },
    {
      id: 'actions', // [!code highlight]
      cell: ({ row }) => {
        return (
          <>
            <Button>Edit</Button>
            <Button>Delete</Button>
          </>
        );
      },
    },
  ]}
  defaultSorting={{
    sortBy: 'id',
    sortDirection: 'desc',
  }}
/>
```

## Usage with Pagination API

You can read how to implement [Pagination API here](/docs/dev/database-pagination).

To help you with the implementation, we created a helper function `getPaginationTool()` that you can use to convert the data from the database to the format that the `DataTable` component expects.

import { Step, Steps } from 'fumadocs-ui/components/steps';
 
<Steps>
<Step>
 
### Get Data

Use the `fetcher` function to get data from the API. You can read more [here about fetch data](/docs/dev/fetch).

```tsx title="example-page.tsx"
import { fetcher } from 'vitnode-frontend/graphql/fetcher';

const getData = async (variables: Test_Query__ShowQueryVariables) => {
  const data = await fetcher<
    Test_Query__ShowQuery,
    Test_Query__ShowQueryVariables
  >({
    query: Test_Query__Show,
    variables,
  });

  return data;
};
```

</Step>
<Step>
 
### Get Search Params

Create new interface for page props.

```tsx title="example-page.tsx"
import { SearchParamsPagination } from 'vitnode-frontend/graphql/get-pagination-tool';

export interface ExamplePageProps {
  searchParams: SearchParamsPagination;
}
```

</Step>
 
<Step>
 
### Provide Interface to Component

```tsx title="example-page.tsx"
export default async function Page({ searchParams }: ExamplePageProps) {}
```

</Step>

<Step>
 
### Use getPaginationTool

```tsx title="example-page.tsx"
import {
  getPaginationTool, // [!code highlight]
  SearchParamsPagination,
} from 'vitnode-frontend/graphql/get-pagination-tool';

export default async function Page({ searchParams }: ExamplePageProps) {
  // [!code word:getPaginationTool]
  const variables = getPaginationTool({
    searchParams,
    defaultPageSize: 10,
  });
}
```

This function will automatically handle the pagination, search and sorting for you when you're using the `DataTable` component.

</Step>
</Steps>
