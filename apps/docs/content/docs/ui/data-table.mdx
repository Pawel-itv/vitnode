---
title: Data Table
description: Powerful table compatible with SSR, search, pagination and more.
---

## Usage

```tsx
import { DataTable } from 'vitnode-frontend/components/ui/data-table';
```

```tsx
<DataTable
  data={[
    { id: 1, name: 'John Doe' },
    { id: 2, name: 'Jane Doe 2' },
  ]}
  columns={[
    { title: 'ID', id: 'id' },
    { title: 'Name', id: 'name' },
  ]}
  defaultSorting={{
    sortBy: 'id',
    sortDirection: 'desc',
  }}
/>
```

## Examples

### With Pagination

```tsx
<DataTable
  data={[
    { id: 1, name: 'John Doe' },
    { id: 2, name: 'Jane Doe 2' },
  ]}
  columns={[
    { title: 'ID', id: 'id' },
    { title: 'Name', id: 'name' },
  ]}
  defaultSorting={{
    sortBy: 'id',
    sortDirection: 'desc',
  }}
  // [!code word:pageInfo]
  pageInfo={{
    count: 2,
    endCursor: 2,
    hasNextPage: false,
    hasPreviousPage: false,
    startCursor: 1,
    totalCount: 2,
  }}
/>
```

### With Search

```tsx
<DataTable
  data={[
    { id: 1, name: 'John Doe' },
    { id: 2, name: 'Jane Doe 2' },
  ]}
  columns={[
    { title: 'ID', id: 'id' },
    { title: 'Name', id: 'name' },
  ]}
  defaultSorting={{
    sortBy: 'id',
    sortDirection: 'desc',
  }}
  searchPlaceholder="Search users..." // [!code highlight]
/>
```

### With Custom Cell

```tsx
<DataTable
  data={[
    { id: 1, name: 'John Doe' },
    { id: 2, name: 'Jane Doe 2' },
  ]}
  columns={[
    { title: 'ID', id: 'id' },
    {
      title: 'Name',
      id: 'name',
      // [!code word:cell]
      cell: ({ row }) => {
        return <span className="font-bold">{row.name}</span>;
      },
    },
  ]}
  defaultSorting={{
    sortBy: 'id',
    sortDirection: 'desc',
  }}
/>
```

#### Actions

We created special column for actions. You can add buttons, links or any other components. Flex is used for the layout with gap.

```tsx
<DataTable
  data={[
    { id: 1, name: 'John Doe' },
    { id: 2, name: 'Jane Doe 2' },
  ]}
  columns={[
    { title: 'ID', id: 'id' },
    { title: 'Name', id: 'name' },
    {
      id: 'actions', // [!code highlight]
      cell: ({ row }) => {
        return (
          <>
            <Button>Edit</Button>
            <Button>Delete</Button>
          </>
        );
      },
    },
  ]}
  defaultSorting={{
    sortBy: 'id',
    sortDirection: 'desc',
  }}
/>
```

### With Sorting Column

```tsx
<DataTable
  data={[
    { id: 1, name: 'John Doe' },
    { id: 2, name: 'Jane Doe 2' },
  ]}
  columns={[
    { title: 'ID', id: 'id' },
    {
      title: 'Name',
      id: 'name',
      sortable: true, // [!code highlight]
    },
  ]}
  defaultSorting={{
    sortBy: 'id',
    sortDirection: 'desc',
  }}
/>
```
