title: Date

## Usage

```tsx
import { DateFormat } from 'vitnode-frontend/components/date-format';
```

```tsx
<DateFormat date={data.joined} />
```

## API Reference

import { TypeTable } from 'fumadocs-ui/components/type-table';

<TypeTable
  type={{
    date: {
      description: 'The date object to display.',
      type: 'Date',
    },
    showFullDate: {
      description: 'Show full date without relative time.',
      type: 'boolean',
      default: 'false',
    },
  }}
/>
